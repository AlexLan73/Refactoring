35int discr_is_monotonic(double A1, double A2, double A3)
36{
37 /*
38 * Монотонные = все убывают ИЛИ все возрастают.
39 * Это значит, что среди A1, A2, A3 нет локального максимума,
40 * т.е. пик ДН находится за пределами сетки из 3 отсчётов.
41 *
42 * Примеры:
43 * A1=0.8, A2=0.6, A3=0.3 → убывают → монотонные (пик слева)
44 * A1=0.3, A2=0.6, A3=0.8 → возрастают → монотонные (пик справа)
45 * A1=0.6, A2=0.9, A3=0.7 → перегиб → НЕ монотонные (пик внутри)
46 */
47 return (A1 >= A2 && A2 >= A3) || /* убывают (или равны) */
48 (A1 <= A2 && A2 <= A3); /* возрастают (или равны) */
49}

58typedef struct {
59 double a;
60 double x;
61} ax_pair_t;

63static int cmp_by_x(const void *p1, const void *p2)
64{
65 const ax_pair_t *a = (const ax_pair_t *)p1;
66 const ax_pair_t *b = (const ax_pair_t *)p2;
67 if (a->x < b->x) return -1;
68 if (a->x > b->x) return 1;
69 return 0;
70}

72int discr3_extrap_gauss(double A1, double A2, double A3,
73 double x1, double x2, double x3,
74 double *xe)
75{
76 ax_pair_t pts[3];
77 double a1, a2, a3, f1, f2, f3;
78 double z1, z2, z3;
79 double a_coef, b_coef;
80 double span;
81
82 /*
83 * Шаг 1. Проверка: все амплитуды > 0
84 * (нужно для логарифмирования)
85 */
86 if (A1 < DBL_EPSILON || A2 < DBL_EPSILON || A3 < DBL_EPSILON) {
87 *xe = x2;
88 return EXIT_FAILURE;
89 }
90
91 /* Все амплитуды одинаковы → невозможно определить направление */
92 if (fabs(A1 - A2) < DBL_EPSILON && fabs(A2 - A3) < DBL_EPSILON) {
93 *xe = x2;
94 return EXIT_FAILURE;
95 }
96
97 /*
98 * Шаг 2. Сортировка по координатам (x возрастает)
99 * Это нужно чтобы формулы работали корректно
100 * независимо от порядка входных аргументов.
101 */
102 pts[0].a = A1; pts[0].x = x1;
103 pts[1].a = A2; pts[1].x = x2;
104 pts[2].a = A3; pts[2].x = x3;
105 qsort(pts, 3, sizeof(ax_pair_t), cmp_by_x);
106
107 a1 = pts[0].a; f1 = pts[0].x;
108 a2 = pts[1].a; f2 = pts[1].x;
109 a3 = pts[2].a; f3 = pts[2].x;
110
111 /*
112 * Шаг 3. НЕ проверяем выпуклость!
113 *
114 * Это ГЛАВНОЕ отличие от discr3ea:
115 * - discr3ea проверяет что a2 > a1 и a2 > a3 (перегиб)
116 * и возвращает FAILURE для монотонных данных
117 * - Мы РАЗРЕШАЕМ монотонные данные, потому что
118 * вершина параболы в log-пространстве покажет
119 * КУДА и НАСКОЛЬКО далеко экстраполировать пик.
120 */
121
122 /*
123 * Шаг 4. Логарифмирование
124 *
125 * Гауссиан y = A*exp(-a*(x-x0)^2) после логарифмирования
126 * становится параболой: ln(y) = ln(A) - a*(x-x0)^2
127 *
128 * Парабола в log-пространстве → вершина = x0 (положение пика).
129 */
130 z1 = log(a1);
131 z2 = log(a2);
132 z3 = log(a3);
133
134 /*
135 * Шаг 5. Коэффициенты параболы
136 *
137 * Парабола z(x) через 3 точки (f1,z1), (f2,z2), (f3,z3):
138 * a_coef = z1*(f2²-f3²) + z2*(f3²-f1²) + z3*(f1²-f2²)
139 * b_coef = z1*(f2-f3) + z2*(f3-f1) + z3*(f1-f2)
140 *
141 * Вершина: xe = 0.5 * a_coef / b_coef
142 */
143 a_coef = z1 * (f2 * f2 - f3 * f3)
144 + z2 * (f3 * f3 - f1 * f1)
145 + z3 * (f1 * f1 - f2 * f2);
146
147 b_coef = z1 * (f2 - f3)
148 + z2 * (f3 - f1)
149 + z3 * (f1 - f2);
150
151 /* b_coef ≈ 0 → парабола вырожденная, направление неопределено */
152 if (fabs(b_coef) < DBL_EPSILON) {
153 /* Экстраполируем в сторону наибольшей амплитуды */
154 if (a3 > a1)
155 *xe = f3 + 0.5 * (f3 - f1);
156 else
157 *xe = f1 - 0.5 * (f3 - f1);
158 return EXIT_FAILURE;
159 }
160
161 /* Вершина параболы в log-пространстве = оценка положения пика */
162 *xe = 0.5 * a_coef / b_coef;
163
164 /*
165 * Шаг 6. Ограничение вылета (clipping)
166 *
167 * Расширенный диапазон: ±2 × размер_сетки (вместо ±0.5× в discr3ea).
168 * Это позволяет экстраполировать дальше, но не бесконечно далеко.
169 *
170 * Почему 2×? Из исследования: при вылете > 2× шага ошибка
171 * экстраполяции уже не уменьшается, а растёт.
172 */
173 span = f3 - f1; /* = 2 × шаг_сетки для симметричной сетки */
174
175 if (*xe > f3 + span) {
176 *xe = f3 + span;
177 return EXIT_FAILURE;
178 }
179 if (*xe < f1 - span) {
180 *xe = f1 - span;
181 return EXIT_FAILURE;
182 }
183
184 return EXIT_SUCCESS;
185}

188/* discr3_auto — автоматический дискриминатор */
189/* ═══════════════════════════════════════════════════════════════════ */
190
191int discr3_auto(double A1, double A2, double A3,
192 double x1, double x2, double x3,
193 double *xe)
194{
195 int ret;
196
197 /*
198 * Ветка 1: МОНОТОННЫЕ данные (пик за сеткой)
199 *
200 * Когда A1 ≥ A2 ≥ A3 или A1 ≤ A2 ≤ A3, среди 3 отсчётов
201 * нет перегиба — пик ДН находится за пределами сетки.
202 *
203 * Стандартные дискриминаторы в этом случае:
204 * CG: MAE = 0.46 (почти случайная оценка)
205 * QA: MAE = 0.32, MaxErr = 1.23 (парабола улетает!)
206 * EA: возвращает FAILURE, ставит xe = краевой отсчёт
207 *
208 * Гауссова экстраполяция (E2): MAE = 0.148
209 * → в 2-3 раза лучше любого стандартного метода!
210 */
211 if (discr_is_monotonic(A1, A2, A3)) {
212 ret = discr3_extrap_gauss(A1, A2, A3, x1, x2, x3, xe);
213 return (ret == EXIT_SUCCESS) ? 2 : 3;
214 /* 2 = E2 сработала, 3 = E2 не смогла (нули, вырожденность) */
215 }
216
217 /*
218 * Ветка 2: НОРМАЛЬНЫЕ данные (пик внутри сетки)
219 *
220 * EA (экспоненциальный): MAE = 0.005 в нормальной зоне
221 * Лучший метод! Использует log-параболу, точнее QA.
222 *
223 * Если EA вернёт FAILURE (вогнутость, нули), переходим на QA.
224 * QA: MAE = 0.006 — почти так же точен, но без ограничений
225 * на знак амплитуд и выпуклость.
226 */
227 ret = discr3ea(A1, A2, A3, x1, x2, x3, xe);
228
229 if (ret == EXIT_SUCCESS) {
230 return 0; /* EA справился — лучший результат */
231 }
232
233 /*
234 * Ветка 3: FALLBACK на QA
235 *
236 * EA вернул FAILURE. Причины:
237 * - Вогнутость (минимум в центре вместо максимума)
238 * - Нулевые или отрицательные амплитуды
239 * - Числовая вырожденность
240 *
241 * QA не требует положительных амплитуд и не проверяет
242 * выпуклость — работает всегда (кроме совсем вырожденных случаев).
243 */
244 *xe = discr3qa(A1, A2, A3, x1, x2, x3);
245 return 1; /* QA fallback */
246}
