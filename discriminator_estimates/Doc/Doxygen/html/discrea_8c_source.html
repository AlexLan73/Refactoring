50int discr3ea(double A1, double A2, double A3,
51 double x1, double x2, double x3, double *xe)
52{
53 double a = 0.0, b = 0.0;
54 double z1, z2, z3;
55 double f1, f2, f3;
56 double a1, a2, a3, a12, a23, a13;
57 /* ERR-002 fix: локальная переменная вместо static */
58 axstr_t ax[3];
59
60 /* Проверка: все амплитуды должны быть положительны */
61 if (A1 < DBL_EPSILON || A2 < DBL_EPSILON || A3 < DBL_EPSILON) {
62 *xe = x2;
63 return EXIT_FAILURE;
64 }
65
66 /* Проверка: не все амплитуды равны */
67 if ((fabs(A1 - A2) < DBL_EPSILON) && (fabs(A2 - A3) < DBL_EPSILON)) {
68 *xe = x2;
69 return EXIT_FAILURE;
70 }
71
72 /* Сортировка по координате (x должна возрастать) */
73 ax[0].a = A1; ax[0].x = x1;
74 ax[1].a = A2; ax[1].x = x2;
75 ax[2].a = A3; ax[2].x = x3;
76 qsort(ax, 3, sizeof(axstr_t), compare_by_x);
77
78 a1 = ax[0].a; a2 = ax[1].a; a3 = ax[2].a;
79 f1 = ax[0].x; f2 = ax[1].x; f3 = ax[2].x;
80
81 /* Проверка на выпуклость: максимум должен быть в центре */
82 a12 = a1 - a2;
83 a23 = a2 - a3;
84 a13 = a1 - a3;
85
86 /* Вогнутость: максимум справа (например 2, 4, 8) */
87 if ((a12 < 0 && a23 < 0 && a12 > a23) ||
88 (a12 > 0 && a23 < 0 && a13 < 0)) {
89 *xe = x3;
90 return EXIT_FAILURE;
91 }
92 /* Вогнутость: максимум слева (например 8, 4, 2) */
93 if ((a12 > 0 && a23 > 0 && a12 > a23) ||
94 (a12 > 0 && a23 < 0 && a13 > 0)) {
95 *xe = x1;
96 return EXIT_FAILURE;
97 }
98
99 /* Логарифмирование */
100 z1 = log(a1);
101 z2 = log(a2);
102 z3 = log(a3);
103
104 /* Коэффициенты параболы z(x) = a + b*x + ... */
105 a = z1 * (f2 * f2 - f3 * f3) + z2 * (f3 * f3 - f1 * f1) + z3 * (f1 * f1 - f2 * f2);
106 b = z1 * (f2 - f3) + z2 * (f3 - f1) + z3 * (f1 - f2);
107
108 if (fabs(b) < DBL_EPSILON) {
109 if (a * b >= 0) {
110 *xe = f3 + 0.5 * (f3 - f1);
111 return EXIT_FAILURE;
112 } else {
113 *xe = f1 - 0.5 * (f3 - f1);
114 return EXIT_FAILURE;
115 }
116 }
117
118 /* Вершина параболы */
119 *xe = 0.5 * a / b;
120
121 /* Ограничение на вылет за пределы зондирования */
122 if (*xe > f3 + 0.5 * (f3 - f1)) {
123 *xe = f3 + 0.5 * (f3 - f1);
124 return EXIT_FAILURE;
125 }
126 if (*xe < f1 - 0.5 * (f3 - f1)) {
127 *xe = f1 - 0.5 * (f3 - f1);
128 return EXIT_FAILURE;
129 }
130
131 return EXIT_SUCCESS;
132}

146double discr3eaY(double A1, double A2, double x1, double x2, double xe)
147{
148 double ye, yMax;
149 double z1, z2;
150 double a0, dxe;
151
152 if (A1 < DBL_EPSILON || A2 < DBL_EPSILON)
153 return 0.0;
154
155 yMax = (A1 > A2) ? A1 : A2;
156
157 z1 = log(A1);
158 z2 = log(A2);
159
160 a0 = (2.0 * xe * (x1 - x2) - x1 * x1 + x2 * x2);
161
162 if (fabs(a0) < DBL_EPSILON)
163 return yMax;
164
165 a0 = (z1 - z2) / a0;
166
167 dxe = x2 - xe;
168 ye = A2 * exp(a0 * dxe * dxe);
169
170 return ye;
171}
