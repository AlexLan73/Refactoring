43int discr2(double maz[2][2], double mel[2][2], double mval[2][2],
44 double *az, double *el, double *val)
45{
46 int maxRow = 0, maxCol = 0;
47 double maxAmpl = -1.0;
48 int i, j;
49
50 /* Проверка: все амплитуды должны быть положительны */
51 for (i = 0; i < 2; ++i)
52 for (j = 0; j < 2; ++j)
53 if (mval[i][j] < DBL_EPSILON) {
54 *az = 0.0; *el = 0.0; *val = 0.0;
55 return -1;
56 }
57
58 /* Поиск максимума */
59 for (i = 0; i < 2; ++i)
60 for (j = 0; j < 2; ++j)
61 if (mval[i][j] > maxAmpl) {
62 maxRow = i;
63 maxCol = j;
64 maxAmpl = mval[i][j];
65 }
66
67 /* Оценка по азимуту (по строке maxRow) */
68 *az = discr2cg(mval[maxRow][0], mval[maxRow][1],
69 maz[maxRow][0], maz[maxRow][1]);
70
71 /* Оценка по углу места (по столбцу maxCol) */
72 *el = discr2cg(mval[0][maxCol], mval[1][maxCol],
73 mel[0][maxCol], mel[1][maxCol]);
74
75 *val = maxAmpl;
76 return 0;
77}

101int discr3(double **maz, double **mel, double **mval, int ndnx, int ndny,
102 DISCRTYPE discrtype, double dx, double lambda,
103 double *az, double *el, double *val)
104{
105 double aztmp = 0.0, eltmp = 0.0, valaz = 0.0, valel = 0.0;
106 int maxRow = 0, maxCol = 0;
107 double maxAmpl = -1.0;
108 const double c = 2.0 * M_PI * dx / lambda;
109 int i, j;
110
111 /* Проверка на возможность использования 3-точечного дискриминатора */
112 if (!checkuse3d(maz, mel, mval, ndnx, ndny))
113 return -1;
114
115 /* Поиск строки и столбца с максимальной амплитудой */
116 for (i = 0; i < ndnx; ++i)
117 for (j = 0; j < ndny; ++j)
118 if (mval[i][j] > maxAmpl) {
119 maxRow = i;
120 maxCol = j;
121 maxAmpl = mval[i][j];
122 }
123
124 switch (discrtype) {
125 case DT_CG:
126 /* ERR-006 fix: добавлена поддержка центра тяжести */
127 if (ndny >= 3) {
128 aztmp = discr3cg(mval[maxRow][0], mval[maxRow][1], mval[maxRow][2],
129 maz[maxRow][0], maz[maxRow][1], maz[maxRow][2]);
130 }
131 if (ndnx >= 3) {
132 eltmp = discr3cg(mval[0][maxCol], mval[1][maxCol], mval[2][maxCol],
133 mel[0][maxCol], mel[1][maxCol], mel[2][maxCol]);
134 }
135 *val = maxAmpl;
136 break;
137
138 case DT_QA:
139 /* ERR-003 fix: координаты умножаем на c локально, не модифицируя входные данные */
140 aztmp = discr3qa(mval[maxRow][0], mval[maxRow][1], mval[maxRow][2],
141 maz[maxRow][0] * c, maz[maxRow][1] * c, maz[maxRow][2] * c);
142 eltmp = discr3qa(mval[0][maxCol], mval[1][maxCol], mval[2][maxCol],
143 mel[0][maxCol] * c, mel[1][maxCol] * c, mel[2][maxCol] * c);
144 *val = maxAmpl;
145 /* Обратное преобразование из масштабированных координат */
146 *az = aztmp / c;
147 *el = eltmp / c;
148 return 0;
149
150 case DT_EA:
151 /* ERR-003 fix: координаты умножаем на c локально */
152 discr3ea(mval[maxRow][0], mval[maxRow][1], mval[maxRow][2],
153 maz[maxRow][0] * c, maz[maxRow][1] * c, maz[maxRow][2] * c, &aztmp);
154 discr3ea(mval[0][maxCol], mval[1][maxCol], mval[2][maxCol],
155 mel[0][maxCol] * c, mel[1][maxCol] * c, mel[2][maxCol] * c, &eltmp);
156
157 valaz = discr3eaY(mval[maxRow][0], mval[maxRow][1],
158 maz[maxRow][0] * c, maz[maxRow][1] * c, aztmp);
159 valel = discr3eaY(mval[0][maxCol], mval[1][maxCol],
160 mel[0][maxCol] * c, mel[1][maxCol] * c, eltmp);
161
162 *val = (valaz > valel) ? valaz : valel;
163 *val = (maxAmpl > *val) ? maxAmpl : *val;
164 /* Обратное преобразование */
165 *az = aztmp / c;
166 *el = eltmp / c;
167 return 0;
168
169 default:
170 *az = 0.0; *el = 0.0; *val = 0.0;
171 return -2;
172 }
173
174 /* Для DT_CG — координаты не масштабируются */
175 *az = aztmp;
176 *el = eltmp;
177 return 0;
178}

249int conv3to2(double maz3[3][3], double mel3[3][3], double mval3[3][3],
250 double maz2[2][2], double mel2[2][2], double mval2[2][2])
251{
252 int bestR = 0, bestC = 0;
253 double bestSum = -1.0;
254 int r, c, i, j;
255
256 /* Перебираем все 4 квадранта 2x2 (смещения: (0,0), (0,1), (1,0), (1,1)) */
257 for (r = 0; r < 2; r++) {
258 for (c = 0; c < 2; c++) {
259 double sum = mval3[r][c] + mval3[r][c + 1]
260 + mval3[r + 1][c] + mval3[r + 1][c + 1];
261 if (sum > bestSum) {
262 bestSum = sum;
263 bestR = r;
264 bestC = c;
265 }
266 }
267 }
268
269 /* Копируем выбранный квадрант */
270 for (i = 0; i < 2; i++) {
271 for (j = 0; j < 2; j++) {
272 maz2[i][j] = maz3[bestR + i][bestC + j];
273 mel2[i][j] = mel3[bestR + i][bestC + j];
274 mval2[i][j] = mval3[bestR + i][bestC + j];
275 }
276 }
277
278 return 0;
279}
