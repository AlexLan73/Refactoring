Генерация тестового сигнала и вычисление оконного FFT.

import numpy as np

try:
    from scipy.signal.windows import blackman, kaiser, flattop, nuttall
    _SCIPY_OK = True
except ImportError:
    _SCIPY_OK = False

def get_window(name: str, N: int) -> np.ndarray:
    """
    Вернуть оконную функцию по имени.

    Встроенные (numpy): ``"hamming"``, ``"hann"``, ``"bartlett"``, ``"blackman"``
    Scipy (если установлен): ``"kaiser8"``, ``"kaiser14"``, ``"flattop"``, ``"nuttall"``

    Parameters
    ----------
    name : str
        Название окна (регистронезависимо)
    N : int
        Длина окна

    Returns
    -------
    np.ndarray, shape (N,)
    """
    name = name.lower()
    if name == "hamming":
        return np.hamming(N)
    if name in ("hann", "hanning"):
        return np.hanning(N)
    if name == "bartlett":
        return np.bartlett(N)
    if name == "blackman":
        return np.blackman(N)
    if name == "rectangular" or name == "rect":
        return np.ones(N)
    # scipy-зависимые окна
    if name == "kaiser8":
        if not _SCIPY_OK:
            raise ImportError("scipy не установлен: pip install scipy")
        return kaiser(N, beta=8.0)
    if name == "kaiser14":
        if not _SCIPY_OK:
            raise ImportError("scipy не установлен: pip install scipy")
        return kaiser(N, beta=14.0)
    if name == "flattop":
        if not _SCIPY_OK:
            raise ImportError("scipy не установлен: pip install scipy")
        return flattop(N)
    if name == "nuttall":
        if not _SCIPY_OK:
            raise ImportError("scipy не установлен: pip install scipy")
        return nuttall(N)
    raise ValueError(f"Неизвестное окно: '{name}'. "
                     "Доступны: hamming, hann, bartlett, blackman, rect, "
                     "kaiser8, kaiser14, flattop, nuttall")

def create_signal(fsin: float, fd: float, N: int,
                  a: float = 1.0, an: float = 1e-9) -> np.ndarray:
    """
    Создать комплексный экспоненциальный сигнал с шумом.

    Генерирует N отсчётов комплексной огибающей:
        s(t) = a · exp(j·2π·fsin·t) + noise

    Parameters
    ----------
    fsin : float
        Частота сигнала, Гц
    fd : float
        Частота дискретизации, Гц
    N : int
        Количество отсчётов
    a : float
        Амплитуда сигнала (по умолчанию 1.0)
    an : float
        СКО шума (по умолчанию 1e-9 — практически без шума)

    Returns
    -------
    np.ndarray (complex128)
        Комплексная огибающая сигнала, N отсчётов
    """
    t = np.arange(N) / fd
    signal = a * np.exp(1j * 2.0 * np.pi * fsin * t)
    if an > 0:
        rng = np.random.default_rng()
        noise = an / np.sqrt(2.0) * (rng.standard_normal(N) +
                                      1j * rng.standard_normal(N))
        signal = signal + noise
    return signal

def compute_fft(signal: np.ndarray, nfft: int,
                window: str = "hamming") -> np.ndarray:
    """
    Вычислить оконный FFT спектр сигнала.

        FFT = fftshift( fft(signal · win(N), nfft) )

    Parameters
    ----------
    signal : np.ndarray
        Комплексный входной сигнал (N отсчётов)
    nfft : int
        Размер FFT
    window : str
        Имя окна (см. :func:`get_window`). По умолчанию ``"hamming"``.

    Returns
    -------
    np.ndarray (complex128)
        Комплексный спектр (fftshift), nfft точек
    """
    N = len(signal)
    win = get_window(window, N)
    return np.fft.fftshift(np.fft.fft(signal * win, nfft))

def make_freq_axis(nfft: int, fd: float) -> np.ndarray:
    """
    Построить частотную ось FFT после fftshift.

    Parameters
    ----------
    nfft : int
        Размер FFT
    fd : float
        Частота дискретизации, Гц

    Returns
    -------
    np.ndarray
        Частоты, Гц, nfft точек: от -fd/2 до fd/2 - df
    """
    return np.fft.fftshift(np.fft.fftfreq(nfft, 1.0 / fd))
