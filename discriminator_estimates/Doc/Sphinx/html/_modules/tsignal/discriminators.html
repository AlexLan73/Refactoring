"""
Частотные дискриминаторы FFT-спектра.

Точная реализация по C-исходникам из проекта discriminator_estimates
(Refactoring/discriminator_estimates/src/).

Методы:
  lay — 3-точечный Jacobsen: -Re{(Sp-Sm)/(2S0-Sm-Sp)} * fd/N  (fcalcdelay.m)
  exp — 3-точечный, парабола на log|S| с проверками  (discr3ea.c)
  sqr — 3-точечный, парабола по формуле Ao           (discr3qa.c)
  sd  — 2-точечный, суммарно-разностный              (discrsd.c)
  cg  — 2-точечный, центр тяжести                    (discrcg.c)
"""
import numpy as np
from typing import Callable


# --------------------------------------------------------------------------- #
#  Вспомогательные функции                                                     #
# --------------------------------------------------------------------------- #

def _peak_triplet(fft_data: np.ndarray) -> tuple[int, int, int]:
    """Вернуть индексы (k_m, k, k_p) трёх бинов вокруг максимума."""
    k = int(np.argmax(np.abs(fft_data)))
    N = len(fft_data)
    return (k - 1) % N, k, (k + 1) % N


def _top2(fft_data: np.ndarray) -> tuple[int, int]:
    """Вернуть индексы двух бинов с наибольшим модулем (k1=max, k2=second)."""
    mag = np.abs(fft_data)
    idx = np.argsort(mag)
    return int(idx[-1]), int(idx[-2])


# --------------------------------------------------------------------------- #
#  lay — Jacobsen (из fcalcdelay.m)                                            #
# --------------------------------------------------------------------------- #



[документация]
def discriminator_lay(fft_data: np.ndarray,
                      f_axis: np.ndarray) -> tuple[float, float]:
    """
    Метод «Лайонса» — формула Jacobsen из fcalcdelay.m.

    sigmapeak = (ffts(k+1)-ffts(k-1)) / (2*ffts(k)-ffts(k-1)-ffts(k+1))
    f = fmax - real(sigmapeak) * fd / count
    """
    k_m, k, k_p = _peak_triplet(fft_data)
    S_m = fft_data[k_m]
    S_0 = fft_data[k]
    S_p = fft_data[k_p]
    denom = 2.0 * S_0 - S_m - S_p
    if abs(denom) > 1e-60:
        sigmapeak = (S_p - S_m) / denom
        N = len(fft_data)
        fd = (f_axis[1] - f_axis[0]) * N
        f_est = f_axis[k] - np.real(sigmapeak) * fd / N
    else:
        f_est = f_axis[k]
    return f_est, abs(fft_data[k])




# --------------------------------------------------------------------------- #
#  exp — discr3ea.c (парабола на log с проверками)                             #
# --------------------------------------------------------------------------- #



[документация]
def discriminator_exp(fft_data: np.ndarray,
                      f_axis: np.ndarray) -> tuple[float, float]:
    """
    Экспоненциальная 3-точечная интерполяция — из discr3ea.c.

    Алгоритм:
      1. Сортировка 3 точек по x (координате)
      2. Проверка на выпуклость (максимум в центре)
      3. Логарифмирование: z = ln(A)
      4. Парабола z(x), вершина: xe = 0.5 * a / b
      5. Ограничение на вылет за пределы ±1.5·(f3-f1) от краёв
    """
    k_m, k, k_p = _peak_triplet(fft_data)
    amp_k = abs(fft_data[k])
    eps = 1e-300

    # Собираем 3 точки (амплитуда, частота)
    points = [
        (abs(fft_data[k_m]), f_axis[k_m]),
        (abs(fft_data[k]),   f_axis[k]),
        (abs(fft_data[k_p]), f_axis[k_p]),
    ]

    A1, A2, A3 = points[0][0], points[1][0], points[2][0]

    # Проверка: все амплитуды > 0
    if A1 < eps or A2 < eps or A3 < eps:
        return f_axis[k], amp_k

    # Проверка: не все равны
    if abs(A1 - A2) < eps and abs(A2 - A3) < eps:
        return f_axis[k], amp_k

    # Сортировка по координате (возрастание x)
    points.sort(key=lambda p: p[1])
    a1, f1 = points[0]
    a2, f2 = points[1]
    a3, f3 = points[2]

    # Проверка на выпуклость
    d12 = a1 - a2
    d23 = a2 - a3
    d13 = a1 - a3
    # Вогнутость: максимум справа
    if (d12 < 0 and d23 < 0 and d12 > d23) or (d12 > 0 and d23 < 0 and d13 < 0):
        return f3, amp_k
    # Вогнутость: максимум слева
    if (d12 > 0 and d23 > 0 and d12 > d23) or (d12 > 0 and d23 < 0 and d13 > 0):
        return f1, amp_k

    # Логарифмирование
    z1, z2, z3 = np.log(a1), np.log(a2), np.log(a3)

    # Коэффициенты параболы
    a_coeff = z1 * (f2**2 - f3**2) + z2 * (f3**2 - f1**2) + z3 * (f1**2 - f2**2)
    b_coeff = z1 * (f2 - f3) + z2 * (f3 - f1) + z3 * (f1 - f2)

    if abs(b_coeff) < 1e-300:
        return f_axis[k], amp_k

    # Вершина параболы
    xe = 0.5 * a_coeff / b_coeff

    # Ограничение на вылет
    span = f3 - f1
    if xe > f3 + 0.5 * span:
        xe = f3 + 0.5 * span
    if xe < f1 - 0.5 * span:
        xe = f1 - 0.5 * span

    return xe, amp_k




# --------------------------------------------------------------------------- #
#  sqr — discr3qa.c (парабола по формуле Ao)                                   #
# --------------------------------------------------------------------------- #



[документация]
def discriminator_sqr(fft_data: np.ndarray,
                      f_axis: np.ndarray) -> tuple[float, float]:
    """
    Квадратичная 3-точечная интерполяция — из discr3qa.c.

    Формула (через отношение Ao):
        Ao = (A2 - A1) / (A2 - A3)
        xe = 0.5 * ((Ao-1)*x2² - Ao*x3² + x1²) / ((Ao-1)*x2 - Ao*x3 + x1)

    Граничные случаи обрабатываются как в оригинале.
    """
    k_m, k, k_p = _peak_triplet(fft_data)
    A1 = abs(fft_data[k_m])
    A2 = abs(fft_data[k])
    A3 = abs(fft_data[k_p])
    x1 = f_axis[k_m]
    x2 = f_axis[k]
    x3 = f_axis[k_p]
    eps = 1e-300

    # Граничный случай: A2 ≈ A3
    if abs(A2 - A3) < eps:
        if abs(A2 - A1) < eps:
            return x2, A2
        if A1 > A2:
            return x1, A2
        return 0.5 * (x3 + x2), A2

    # Граничный случай: A1 ≈ A2
    if abs(A2 - A1) < eps:
        if A3 > A2:
            return x3, A2
        return 0.5 * (x1 + x2), A2

    # Основная формула
    Ao = (A2 - A1) / (A2 - A3)
    denom = (Ao - 1.0) * x2 - Ao * x3 + x1
    if abs(denom) < eps:
        return x2, A2
    xe = 0.5 * ((Ao - 1.0) * x2**2 - Ao * x3**2 + x1**2) / denom
    return xe, A2




# --------------------------------------------------------------------------- #
#  sd — discrsd.c (2-точечный суммарно-разностный)                             #
# --------------------------------------------------------------------------- #



[документация]
def discriminator_sd(fft_data: np.ndarray,
                     f_axis: np.ndarray) -> tuple[float, float]:
    """
    2-точечный суммарно-разностный — из discrsd.c.

    Из fcalcdelay.m:
        f = discr2sd(0.132497, P_max^2, P_second^2, f_max, f_second)

    Из discrsd.c:
        xc = (x1 + x2) / 2
        dx = c * (A2 - A1) / (A2 + A1)
        xe = xc + dx
    """
    KSD = 0.132497
    k1, k2 = _top2(fft_data)
    A1 = abs(fft_data[k1]) ** 2   # мощность первого (макс.)
    A2 = abs(fft_data[k2]) ** 2   # мощность второго
    x1 = f_axis[k1]
    x2 = f_axis[k2]
    total = A2 + A1
    if abs(total) < 1e-300:
        return (x1 + x2) * 0.5, abs(fft_data[k1])
    xc = (x1 + x2) * 0.5
    dx = KSD * (A2 - A1) / total
    return xc + dx, abs(fft_data[k1])




# --------------------------------------------------------------------------- #
#  cg — discrcg.c (2-точечный центр тяжести)                                  #
# --------------------------------------------------------------------------- #



[документация]
def discriminator_cg(fft_data: np.ndarray,
                     f_axis: np.ndarray) -> tuple[float, float]:
    """
    2-точечный центр тяжести — из discrcg.c.

    Из fcalcdelay.m:
        f = discr2cg(a_max, a_second, f_max, f_second)

    Из discrcg.c:
        xe = (A1*x1 + A2*x2) / (A1 + A2)
    """
    k1, k2 = _top2(fft_data)
    A1 = abs(fft_data[k1])
    A2 = abs(fft_data[k2])
    x1 = f_axis[k1]
    x2 = f_axis[k2]
    total = A1 + A2
    if abs(total) < 1e-300:
        return (x1 + x2) * 0.5, A1
    return (A1 * x1 + A2 * x2) / total, A1
